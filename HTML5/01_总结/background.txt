			/*background-image: 背景图片*/
			background-image: url(img/pig1.png);
			/*background-repeat平铺：
			（1）no-repeat不平铺
			（2）repeat-x横向平铺
			（3）repeat-y纵向平铺*/
			background-repeat: no-repeat;
			/*background-size: 背景图片尺寸
			（1）横向尺寸
			（2）纵向尺寸
			值可以为px、em、百分比*/
			/*background-size: 100% 100%;*/
			background-size: 2em 2em;
			/*backgroung-position: 背景图片位置
			写法：
			（1）横向位置 纵向位置px、
				 百分比（尺寸百分百不能设置）
			（2）设置方位 top/bottom/right/left/center*/
			background-position: 50px 50px;
			/*background-position: 30% 30%;*/
			background-position: center right;
			
			
			background-attachment 设置背景是否随页面滚动
			（1）scroll 随页面滚动
			（2）fixed 不随页面滚动
			
			
			注释：可以直接采用复合写法
			background:颜色 图片路径 定位 平铺等
			他们之间的顺序可以变化的，但是不推荐，要养成习惯
			例子：background:#f42304 url("../img/1.jpg") 10px 100px no-repeat;
			 
			 当使用简写属性时，属性值的顺序为：:
				background-color
				background-image
				background-repeat
				background-attachment
				background-position
